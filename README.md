```shell (for MAC)
# create env
$ python3 -m venv venv

# activate env 
$ source venv/bin/activate

# install 
$ pip3 install -r requirements.txt

shell (for Windows)
# install virtualenv
pip install virtualenv

# create env
virtualenv venv

# activate env 
.\venv\Scripts\activate

# install 
pip install -r requirements.txt

# run script
python main.py
```
